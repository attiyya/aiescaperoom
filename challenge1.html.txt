    <script>
        // Function to show custom message box instead of alert
        function showMessageBox(message) {
            const msgBox = document.getElementById('messageBox');
            msgBox.textContent = message;
            msgBox.style.display = 'block';
            setTimeout(() => {
                msgBox.style.display = 'none';
            }, 3000); // Hide after 3 seconds
        }

        function checkAnswer() {
            const userAnswer = document.getElementById('answerInput').value.trim().toUpperCase();
            const correctAnswer = 'INPUT'; // P U T I N -> INPUT
            const feedbackElement = document.getElementById('feedback');

            if (userAnswer === correctAnswer) {
                feedbackElement.textContent = 'Success! Youâ€™ve identified the crucial first step. Your first piece of the final ethical code is: 1-[I].';
                feedbackElement.className = 'text-sm font-medium mt-2 text-green-600';
                document.getElementById('answerInput').disabled = true;
                document.querySelector('.btn').disabled = true;
                showMessageBox('Puzzle Solved! Great job on the data pipeline.');
            } else {
                feedbackElement.textContent = 'Incorrect. The data stream is still scrambled. Try again!';
                feedbackElement.className = 'text-sm font-medium mt-2 text-red-600';
                showMessageBox('Try again!');
            }
        }
